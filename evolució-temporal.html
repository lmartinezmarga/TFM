Es creen taules resum per any, estació, mes i dia de la setmana, amb nombre d’incendis i hectàrees cremades.

taula_any <- incendis %>%
  group_by(ANY) %>%
  summarise(
    n_incendis = n(),
    HA_CREMADES = sum(HA_CREMADES, na.rm = TRUE)
  )

taula_estacio <- incendis %>%
  group_by(TEMPORADA) %>%
  summarise(
    n_incendis = n(),
    HA_CREMADES = sum(HA_CREMADES, na.rm = TRUE)
  )

taula_mes <- incendis %>%
  group_by(MES) %>%
  summarise(
    n_incendis = n(),
    HA_CREMADES = sum(HA_CREMADES, na.rm = TRUE)
  ) %>%
  arrange(MES)

taula_dia_set <- incendis %>%
  group_by(DIA_SETMANA) %>%
  summarise(
    n_incendis = n(),
    HA_CREMADES = sum(HA_CREMADES, na.rm = TRUE)
  )

Es genera un conjunt de gràfics per analitzar l’evolució i distribució dels incendis segons diferents variables temporals.
